######################################################################################
# Build function for communications libraries
######################################################################################

SET(DEBUG_DEFINES FALSE)

IF(WIN32)
  SET(DEBUG_DEFINES "-DFTD2XX_WINDOWS")
ENDIF(WIN32)

IF(LIBUSB_DEBUG)
  SET(DEBUG_DEFINES "-DLIBUSB_DEBUG")
ENDIF(LIBUSB_DEBUG)

IF(LIBUSB_1_FOUND)
  NP_BUILD_LIB(
	NAME ${LIBNIFALCON_CPP_COMM_PREFIX}_libusb 
	SOURCES "FalconCommLibUSB.cpp" 
	CXX_FLAGS ${DEBUG_DEFINES} 
	LINK_LIBS "${LIBUSB_1_LIBRARIES}" 
	LINK_FLAGS FALSE
	DEPENDS FALSE
	)
ENDIF(LIBUSB_1_FOUND)

IF(LIBFTDI_FOUND)
  NP_BUILD_LIB(
	NAME ${LIBNIFALCON_CPP_COMM_PREFIX}_libftdi
	SOURCES "FalconCommLibFTDI.cpp" 
	CXX_FLAGS ${DEBUG_DEFINES} 
	LINK_LIBS "${LIBFTDI_LIBRARIES}" 
	LINK_FLAGS FALSE
	DEPENDS FALSE
	)
ENDIF(LIBFTDI_FOUND)

IF(LIBFTD2XX_FOUND)
  NP_BUILD_LIB(
	NAME ${LIBNIFALCON_CPP_COMM_PREFIX}_ftd2xx
	SOURCES "FalconCommFTD2XX.cpp" 
	CXX_FLAGS ${DEBUG_DEFINES} 
	LINK_LIBS "${LIBFTD2XX_LIBRARIES}" 
	LINK_FLAGS FALSE
	DEPENDS FALSE
	)
ENDIF(LIBFTD2XX_FOUND)
